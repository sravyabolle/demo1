//inner join
select * from emps
inner join 
depts
on
emps.depts id= depts.id;

//left join
select * from emps
left outer join 
depts
on
emps.depts id= depts.id;


//right outer join
select * from emps
right outer join 
depts
on
emps.depts id= depts.id;



//full outer join
select * from emps
full outer join 
depts
on
emps.depts id= depts.id;
